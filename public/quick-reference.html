Quick Example 

// Load courses with caching and filters
async function loadCourses() {
    try {
        const data = await fetchCoursesWithCache({
            search: 'programming',
            major: 1,
            level: 3,
            limit: 12
        });
        
        displayCourses(data.courses, 'courses-grid');
        
        // Prefetch next page
        if (data.pagination.hasMore) {
            prefetchNextPage(12, { search: 'programming' });
        }
        
    } catch (error) {
        showMessage(error.message, 'error');
    }
}

// Check performance
console.log(catalogPerformanceMonitor.getSummary());
